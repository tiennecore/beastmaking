# Plan Sync — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Auto-complete planned sessions when the user logs a matching activity (bloc, voie, hangboard), with 2-step progress for mixed sessions.

**Architecture:** A single `tryAutoCompletePlanSession()` function in `lib/storage.ts` handles all matching logic. Called after every session save (add-bloc, add-voie, recap). `SessionCompletion` extended with `progress` for mixed sessions. Plan UI updated with 3-color checkmark (grey/orange/green).

**Tech Stack:** TypeScript, AsyncStorage, React Native

## Task 2: Create `tryAutoCompletePlanSession()` in storage

**Files:**
- Modify: `lib/storage.ts`

**Step 1: Add the matching function**

Add after the existing plan functions (after `advanceWeek`). The function signature:

```typescript
export type AutoCompleteInput =
  | { type: 'climbing'; climbingType: 'bloc' | 'voie' }
  | { type: 'hangboard'; protocolId: string };

export async function tryAutoCompletePlanSession(input: AutoCompleteInput): Promise<void> {
  const active = await loadActivePlan();
  if (!active) return;

  const week = active.weekHistory.find((w) => w.weekNumber === active.currentWeek);
  if (!week) return;

  const sessions = active.plan.sessions;
  let matched = false;

  for (const session of sessions) {
    // Skip already fully completed sessions
    const completion = week.completions.find((c) => c.sessionId === session.id);
    if (completion?.completed) continue;

    if (input.type === 'climbing') {
      const activityMatch =
        (input.climbingType === 'bloc' && session.climbingActivity === 'bouldering') ||
        (input.climbingType === 'voie' && session.climbingActivity === 'route');

      if (session.mode === 'climbing' && activityMatch) {
        // Simple climbing session — mark complete
        if (!completion) {
          week.completions.push({
            sessionId: session.id,
            completed: true,
            skipped: false,
            completedAt: new Date().toISOString(),
            progress: 2,
          });
        } else {
          completion.completed = true;
          completion.completedAt = new Date().toISOString();
          completion.progress = 2;
        }
        matched = true;
        break;
      }

      if (session.mode === 'climbing-exercise' && activityMatch) {
        // Mixed session — increment progress (climbing part)
        const currentProgress = completion?.progress ?? 0;
        if (currentProgress === 0) {
          // First part done
          if (!completion) {
            week.completions.push({
              sessionId: session.id,
              completed: false,
              skipped: false,
              progress: 1,
            });
          } else {
            completion.progress = 1;
          }
          matched = true;
          break;
        }
        // If progress is already 1 from exercise, complete it
        if (currentProgress === 1) {
          completion!.completed = true;
          completion!.completedAt = new Date().toISOString();
          completion!.progress = 2;
          matched = true;
          break;
        }
      }
    }

    if (input.type === 'hangboard') {
      if (session.mode === 'exercise' && session.protocolIds?.includes(input.protocolId)) {
        // Hangboard-only session — mark complete
        if (!completion) {
          week.completions.push({
            sessionId: session.id,
            completed: true,
            skipped: false,
            completedAt: new Date().toISOString(),
            progress: 2,
          });
        } else {
          completion.completed = true;
          completion.completedAt = new Date().toISOString();
          completion.progress = 2;
        }
        matched = true;
        break;
      }

      if (session.mode === 'climbing-exercise' && session.protocolIds?.includes(input.protocolId)) {
        // Mixed session — increment progress (exercise part)
        const currentProgress = completion?.progress ?? 0;
        if (currentProgress === 0) {
          if (!completion) {
            week.completions.push({
              sessionId: session.id,
              completed: false,
              skipped: false,
              progress: 1,
            });
          } else {
            completion.progress = 1;
          }
          matched = true;
          break;
        }
        if (currentProgress === 1) {
          completion!.completed = true;
          completion!.completedAt = new Date().toISOString();
          completion!.progress = 2;
          matched = true;
          break;
        }
      }
    }
  }

  if (matched) {
    await saveActivePlan(active);
  }
}
```
